#include <stdint.h>
#include <tm4c123gh6pm.h>
void delay(void) {
    volatile uint32_t i;
    for (i = 0; i < 500000; i++);
}
int main(void) {
    SYSCTL_RCGCGPIO_R |= 0x20;  // Enable Port F clock
    GPIO_PORTF_DIR_R |= 0x0E;   // Set PF1-PF3 as outputs
    GPIO_PORTF_DEN_R |= 0x0E;   // Digital enable PF1-PF3
    while (1) {
        GPIO_PORTF_DATA_R ^= 0x02; // Toggle PF1 (Red LED)
        delay();
    }
}
